// 1. 생성자 (여러분 마음대로 여러 개 만들기) 
	
	// 2. getters (형식에 맞게)
	
	// 3. setters (재료를 넣으면 검열을 거쳐 올바른 데이터만 들어감)
	//  0) name : 이름
	//  1) kr, en, ma : 0 점 이상 100점 이하만 저장 가능, 그 외 0점 
	//  2) avg : (인자값 받지 않고, kr, en, ma만 가지고 계산되도록)
	//  3) pass : 60점 이상이면 true
	//  4) grade : ABCDF 중 1개로

package home;
class Student {
	private String name;
	private int kr, en, ma;
	private double avg;
	private boolean pass;
	private String grade;
	
	Student() {
		this(null, 0, 0, 0);
	}
	
	Student(String name) {
		this(name, 0, 0, 0);
	}
	
	Student(String name, int kr, int en, int ma) {
		this.name = name;
		this.kr = kr;
		this.en = en;
		this.ma = ma;
	}
	
	public void setName(String name){
		  	this.name = name;
		   }
	
	public void setKr(int kr){
	  	if(kr>=0 && kr<=100) {
	  		this.kr = kr;
	  	} else {
	  	this.kr = 0; }
	   }
	public void setEn(int en){
	  	if(en>=0 && en<=100) {
	  		this.en = en;
	  	}
	  	else {
		  	this.en = 0; }
	   }
	public void setMa(int ma){
	  	if(ma>=0 && ma<=100) {
	  		this.ma = ma;
	  	}
	  	else {
		  	this.ma = 0; }
	   }
	
	public void setAvg() {
		this.avg = (kr + en + ma) / 3.0;
	}
	
	public void setPass() {
		if(avg>=60) {
			System.out.println("true");
		} else { System.out.println("fail");}
	}
	
	public void setGrade() {
		if (avg >= 90) {
			System.out.println("A");
		} else if (avg >=80) {
			System.out.println("B");
		} else if (avg >=70) {
			System.out.println("C");
		} else if (avg >=60) {
			System.out.println("D");
		} else {
			System.out.println("F");
		}
	}
	
	public String getName(){
		 return name;
		 }
	
	public int getKr(){
		 return kr;
		 }
	
	public int getEn(){
		 return en;
		 }
	
	public int getMa(){
		 return ma;
		 }
	
	public double getAvg(){
		 return avg;
		 }
	
	public boolean isPass(){
		 return pass;
		 }
	
	public String getGrade() {
		return grade;
	}

	
}
public class Quiz01 {
	public static void main(String[] args) {
		// Student 테스트 (마음대로)
		
		Student s = new Student();
		
		s.setName("홍길동");
		s.setKr(90);
		s.setEn(80);
		s.setMa(77);
		s.setAvg();
		s.setPass();
		s.setGrade();
		
		
		System.out.println(s.getName()); 
		System.out.println(s.getKr()); 
		System.out.println(s.getEn()); 
		System.out.println(s.getMa()); 
		System.out.println(s.getAvg());
		System.out.println(s.getGrade());
	}
}
